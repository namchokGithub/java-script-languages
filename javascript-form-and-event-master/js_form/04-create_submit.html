04 - Create Submit

สร้างฟอร์มด้วย JavaScript

Code การสร้างฟอร์ม

let form = document.createElement('form')

form.action='https://google.com/search'

form.method='GET'

form.innerHTML='<input name="q"> <button type="submit">ค้นหาด้วย Google</button>' // name ต้อง q เฉพาะ Google

document.body.append(form)
